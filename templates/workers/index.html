Ubuntu Worker Setup
Run these commands on the Ubuntu box (THOR / binsky)
1 — Install system deps
sudo apt update && sudo apt install -y python3-pip python3-venv git curl
# NVIDIA (skip if no GPU on this box):
# sudo apt install -y nvidia-cuda-toolkit
2 — Create worker directory
mkdir -p ~/hihi-worker && cd ~/hihi-worker
3 — Download bundle from hihilabs
curl -L -o worker.zip \
  -H "X-Worker-Key: <WORKER_SECRET from config.env below>" \
  https://hihilabs.xyz/workers/download/
unzip -o worker.zip && rm worker.zip
4 — Python virtualenv + install
python3 -m venv venv && source venv/bin/activate
pip install -r requirements.txt
5 — Create config.env
cat > config.env << 'EOF'
PLESK_URL=https://hihilabs.xyz/workers
WORKER_SECRET=<ask staff for a valid worker secret_key>
WORKER_NAME=BinskyBox
ANTHROPIC_API_KEY=<your Anthropic API key>
OLLAMA_URL=http://localhost:11434
OLLAMA_AI_MODEL=deepseek-r1:14b
OPEN_WEBUI_URL=http://10.0.0.124:6969
OPEN_WEBUI_KEY=<your Open WebUI API key>
POLL_INTERVAL=10
EOF
6 — Run (or set up as systemd service)
# One-shot test:
source venv/bin/activate && python3 worker.py

# Systemd service (run once):
sudo tee /etc/systemd/system/hihi-worker.service > /dev/null << 'EOF'
[Unit]
Description=HiHi Worker
After=network.target

[Service]
User={{ request.user.username }}
WorkingDirectory=/home/{{ request.user.username }}/hihi-worker
ExecStart=/home/{{ request.user.username }}/hihi-worker/venv/bin/python worker.py
Restart=on-failure
RestartSec=10
EnvironmentFile=/home/{{ request.user.username }}/hihi-worker/config.env

[Install]
WantedBy=multi-user.target
EOF
sudo systemctl daemon-reload
sudo systemctl enable --now hihi-worker
sudo systemctl status hihi-worker
Optional — Ollama on Ubuntu (for local AI)
curl -fsSL https://ollama.com/install.sh | sh
ollama pull deepseek-r1:14b
ollama pull llama3.2-vision:11b   # for OCR tasks
